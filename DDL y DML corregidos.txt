create database Hospital;
use Hospital;
create table Estado(
cod int,
tipo varchar(50),
primary key(cod)
);

create table Sede(
cod int,
consultorio varchar(30),
direccion varchar(30),
primary key(cod)
);

create table Horario_c(
cod int, 
hora_c time,
primary key(cod)
);

create table Cod_tipo_id_paciente(
cod int,
tipo varchar(50),
primary key(cod)
);

create table Municipio(
cod int,
nombre varchar(100),
primary key(cod)
);

create table Paciente(
num_id int,
nombre varchar(100),
fecha_nac date,
cod_tipo_id int,
cod_municipio int,
foreign key(cod_tipo_id)
references Cod_tipo_id_paciente(cod),
foreign key(cod_municipio)
references Municipio(cod),
primary key(num_id)
);

create table Telefono_paciente(
id_paciente int,
foreign key(id_paciente)
references Paciente(num_id),
telefono double,
primary key(id_paciente, telefono)
);

create table Cod_tipo_id_medico(
cod int,
tipo_id varchar(50),
primary key(cod)
);

create table Especialidad(
cod int,
nombre varchar(100),
primary key(cod)
);

create table Medico(
num_id int, 
nombre varchar(100),
direccion varchar(30),
cod_tipo int,
cod_especialidad int,
hora_i time,
hora_f time,
cod_horario int,
cod_municipio int,
foreign key(cod_tipo)
references Cod_tipo_id_medico(cod),
foreign key(cod_especialidad)
references Especialidad(cod),
foreign key(cod_horario)
references Horario_c(cod),
foreign key(cod_municipio)
references Municipio(cod),
primary key(num_id)
);

create table Telefono_medico(
id_medico int,
foreign key(id_medico)
references Medico(num_id),
telefono double,
primary key(id_medico, telefono)
);

create table Cita_medica(
cod int,
cod_estado int,
cod_sede int, 
id_paciente int,
id_medico int,
cod_horario int,
fecha_r date,
foreign key(cod_estado) 
references Estado(cod),
foreign key(cod_sede)
references Sede(cod),
foreign key(id_paciente)
references Paciente(num_id),
foreign key(id_medico)
references Medico(num_id),
foreign key(cod_horario)
references Horario_c(cod),
primary key(cod)
);
